**Goal:** Add a per-company canonical sort order to `Client` and `Project`, expose it through service functions and server actions, and update every Client/Project read to honor it. UI is unchanged in this task — the foundation for [CLIENTS-04](../CLIENTS-04/assignment.md)'s drag-and-drop and the order propagation everywhere lists are rendered.

**Architecture:** Add `sortOrder Int @default(0)` to both Prisma models with composite indexes; extend `AuditAction` with `reorder`; backfill existing rows by current alphabetical order in the migration's data step. Add two service functions (`reorderClients`, `reorderProjects`) following the existing `(db, actorUserId, input) => Result<true>` shape, each writing exactly one audit row via the shared `writeAudit` helper. Sweep three call sites to use the canonical sort tuple `[{ archived: 'asc' }, { sortOrder: 'asc' }, { name: 'asc' }]`. Wire two new server actions over the services.

**Tech Stack:** Prisma 6 + Postgres 16, Next.js 15 server actions, Vitest with testcontainers (`apps/web/tests/services/catalog.test.ts`).

**Spec deviation:** `assignment.md` describes the service signature as `reorderClients({ companyId, actorUserId, orderedIds })`. The codebase's existing service convention is `(db, actorUserId, input)` returning a `Result<T>` (see `services/catalog.ts`, `services/audit.ts`). The plan follows the existing convention: `reorderClients(db, actorUserId, { companyId, orderedIds })` returning `Result<true>`. Same scope, idiomatic shape.

Append the following SQL block at the end of `migration.sql` (after the column-add and enum-add statements):

```sql
-- Backfill clients: alphabetical order within each company
WITH ranked AS (
  SELECT id,
         row_number() OVER (PARTITION BY company_id ORDER BY name ASC) AS rn
  FROM clients
)
UPDATE clients c
SET sort_order = ranked.rn
FROM ranked
WHERE c.id = ranked.id;

-- Backfill projects: alphabetical order within each client
WITH ranked AS (
  SELECT id,
         row_number() OVER (PARTITION BY client_id ORDER BY name ASC) AS rn
  FROM projects
)
UPDATE projects p
SET sort_order = ranked.rn
FROM ranked
WHERE p.id = ranked.id;
```

```ts
export async function reorderClients(
  db: Db,
  actorUserId: string,
  input: { companyId: string; orderedIds: string[] },
): Promise<Result<true>> {
  const auth = await requireAdmin(db, actorUserId, input.companyId);
  if (!auth.ok) return auth;

  const active = await db.client.findMany({
    where: { companyId: input.companyId, archived: false },
    orderBy: { sortOrder: 'asc' },
    select: { id: true },
  });
  const activeIds = new Set(active.map((c) => c.id));
  const requested = new Set(input.orderedIds);
  if (activeIds.size !== requested.size || [...requested].some((id) => !activeIds.has(id))) {
    return { ok: false, reason: 'not_found' };
  }

  const before = active.map((c) => c.id);

  await Promise.all(
    input.orderedIds.map((id, i) =>
      db.client.update({ where: { id }, data: { sortOrder: i + 1 } }),
    ),
  );

  await writeAudit(db, {
    companyId: input.companyId,
    actorUserId,
    action: 'reorder',
    entityType: 'client_order',
    entityId: input.companyId,
    before: { ids: before },
    after: { ids: input.orderedIds },
  });

  return { ok: true, value: true };
}
```

```ts
export async function reorderProjects(
  db: Db,
  actorUserId: string,
  input: { companyId: string; clientId: string; orderedIds: string[] },
): Promise<Result<true>> {
  const client = await db.client.findUnique({
    where: { id: input.clientId },
    select: { companyId: true },
  });
  if (!client || client.companyId !== input.companyId) {
    return { ok: false, reason: 'not_found' };
  }
  const auth = await requireAdmin(db, actorUserId, input.companyId);
  if (!auth.ok) return auth;

  const active = await db.project.findMany({
    where: { clientId: input.clientId, archived: false },
    orderBy: { sortOrder: 'asc' },
    select: { id: true },
  });
  const activeIds = new Set(active.map((p) => p.id));
  const requested = new Set(input.orderedIds);
  if (activeIds.size !== requested.size || [...requested].some((id) => !activeIds.has(id))) {
    return { ok: false, reason: 'not_found' };
  }

  const before = active.map((p) => p.id);

  await Promise.all(
    input.orderedIds.map((id, i) =>
      db.project.update({ where: { id }, data: { sortOrder: i + 1 } }),
    ),
  );

  await writeAudit(db, {
    companyId: input.companyId,
    actorUserId,
    action: 'reorder',
    entityType: 'project_order',
    entityId: input.clientId,
    before: { ids: before },
    after: { ids: input.orderedIds },
  });

  return { ok: true, value: true };
```
