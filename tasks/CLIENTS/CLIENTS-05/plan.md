# CLIENTS-05 — Implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bring Playwright into active use, then write the E2E suite that covers the search and drag-and-drop features delivered by [CLIENTS-03](../CLIENTS-03/assignment.md) and [CLIENTS-04](../CLIENTS-04/assignment.md). Tests run against the same Postgres `apps/web/tests/services/*` use locally (testcontainers) and the GitHub-hosted services in CI.

**Architecture:** A single `playwright.config.ts` with a `globalSetup` hook that boots a clean DB, seeds a fixed admin/company/clients/projects, then writes a Playwright `storageState` file holding the admin's session cookie. Specs reuse the storage state, run against `next start` (booted by Playwright's `webServer` config), and assert on the `/clients` page. Local runs use the existing testcontainers DB via `pnpm db:up`; CI uses the workflow's Postgres service.

**Tech Stack:** Playwright, `@playwright/test`, Next.js 15 production mode (Playwright's `webServer`), Auth.js v5 session table.

**Spec deviation:** The cross-company existence-leak guard is exercised at the service level by [CLIENTS-01](../CLIENTS-01/plan.md) Task 2/3 (`reorderClients/reorderProjects returns not_found`). Adding a parallel E2E test would require a test-only API endpoint that calls the action with a foreign-company id — invasive infrastructure for one assertion. Plan ships the integration-level coverage and notes the trade-off in [`changelog.md`](changelog.md).

**Spec:** [`assignment.md`](assignment.md)

`apps/web/tests/e2e/global-setup.ts`:

```ts
import { mkdir, writeFile } from 'node:fs/promises';
import { join } from 'node:path';
import { PrismaClient } from '@prisma/client';
import { resetDb } from '@tt/db/test';
import { encode } from 'next-auth/jwt';

export const E2E_ADMIN_EMAIL = 'e2e-admin@example.test';

export interface SeededWorld {
  companyId: string;
  adminUserId: string;
  clients: { id: string; name: string }[];
  projects: { id: string; clientId: string; name: string }[];
}

let cached: SeededWorld | undefined;

async function seed(prisma: PrismaClient): Promise<SeededWorld> {
  await resetDb(prisma);

  const admin = await prisma.user.create({
    data: { email: E2E_ADMIN_EMAIL, fullName: 'E2E Admin' },
  });
  const company = await prisma.company.create({
    data: { name: 'E2E Co', slug: `e2e-co-${Date.now()}`, createdById: admin.id },
  });
  await prisma.membership.create({
    data: { userId: admin.id, companyId: company.id, role: 'admin' },
  });

  // Three clients in a known order, including a diacritic name and an archived one.
  const clientA = await prisma.client.create({
    data: { companyId: company.id, name: 'Agent 42', sortOrder: 1 },
  });
  const clientB = await prisma.client.create({
    data: { companyId: company.id, name: 'Agént Diakritika', sortOrder: 2 },
  });
  const clientC = await prisma.client.create({
    data: { companyId: company.id, name: 'Old Co', archived: true, sortOrder: 3 },
  });

  // Projects under clientA in a known order.
  const projectGoogle = await prisma.project.create({
    data: { clientId: clientA.id, name: 'Google Work Space', sortOrder: 1 },
  });
  const projectInstall = await prisma.project.create({
    data: { clientId: clientA.id, name: 'Instalace agenta', sortOrder: 2 },
  });
  const projectVps = await prisma.project.create({
    data: { clientId: clientA.id, name: 'Nastavování VPS', sortOrder: 3 },
  });

  return {
    companyId: company.id,
    adminUserId: admin.id,
    clients: [
      { id: clientA.id, name: clientA.name },
      { id: clientB.id, name: clientB.name },
      { id: clientC.id, name: clientC.name },
    ],
    projects: [
      { id: projectGoogle.id, clientId: clientA.id, name: projectGoogle.name },
      { id: projectInstall.id, clientId: clientA.id, name: projectInstall.name },
      { id: projectVps.id, clientId: clientA.id, name: projectVps.name },
    ],
  };
}

async function buildSessionCookie(user: { id: string; email: string }): Promise<{
  name: string;
  value: string;
  domain: string;
  path: string;
  expires: number;
  httpOnly: boolean;
  secure: boolean;
  sameSite: 'Lax' | 'Strict' | 'None';
}> {
  const secret = process.env.AUTH_SECRET ?? 'e2e-test-secret-do-not-use-in-production';
  const oneWeek = 60 * 60 * 24 * 7;
  const token = await encode({
    token: { sub: user.id, email: user.email, name: 'E2E Admin' },
    secret,
    salt: 'authjs.session-token',
    maxAge: oneWeek,
  });
  return {
    name: 'authjs.session-token',
    value: token,
    domain: 'localhost',
    path: '/',
    expires: Math.floor(Date.now() / 1000) + oneWeek,
    httpOnly: true,
    secure: false,
    sameSite: 'Lax',
  };
}

export default async function globalSetup(): Promise<SeededWorld> {
  const prisma = new PrismaClient();
  await prisma.$connect();
  try {
    cached = await seed(prisma);
    const cookie = await buildSessionCookie({
      id: cached.adminUserId,
      email: E2E_ADMIN_EMAIL,
    });
    const dir = join(__dirname, '.auth');
    await mkdir(dir, { recursive: true });
    await writeFile(
      join(dir, 'admin.json'),
      JSON.stringify(
        {
          cookies: [cookie],
          origins: [],
        },
        null,
        2,
      ),
      'utf8',
    );
    await writeFile(join(dir, 'world.json'), JSON.stringify(cached, null, 2), 'utf8');
  } finally {
    await prisma.$disconnect();
  }
  return cached!;
}
```

The `world.json` file lets specs read seeded IDs without re-querying.

import { readFileSync } from 'node:fs';
import { join } from 'node:path';

interface SeededWorld {
  companyId: string;
  adminUserId: string;
  clients: { id: string; name: string }[];

test.describe('clients page — search', () => {
  test('US-51: typing filters clients and auto-expands clients matched via a project', async ({
    page,
  }) => {
    await page.goto('/clients');
    await expect(page.getByText('Agent 42')).toBeVisible();
    await expect(page.getByText('Agént Diakritika')).toBeVisible();

    const search = page.getByRole('searchbox', { name: /hledat klienta nebo projekt/i });
    await search.fill('instalace');

    await expect(page.getByText('Agent 42')).toBeVisible();
    await expect(page.getByText('Agént Diakritika')).toBeHidden();
    await expect(page.getByText('Instalace agenta')).toBeVisible();
    await expect(page.getByText('Google Work Space')).toBeHidden();
  });

  test('US-51: clearing search collapses auto-expanded clients to manual state', async ({
    page,
  }) => {
    await page.goto('/clients');
    // Manually expand "Agént Diakritika"
    await page.getByRole('button', { name: 'Rozbalit projekty' }).nth(1).click();

    const search = page.getByRole('searchbox', { name: /hledat/i });
    await search.fill('instalace');
    await expect(page.getByText('Instalace agenta')).toBeVisible();

    await search.fill('');
    // Manually-expanded "Agént Diakritika" stays open; auto-expanded "Agent 42" should not stay forced open
    // (its manual state is "closed", so it collapses).
    await expect(page.getByText('Instalace agenta')).toBeHidden();
  });

  test('US-51: search is diacritic-insensitive', async ({ page }) => {
    await page.goto('/clients');
    const search = page.getByRole('searchbox', { name: /hledat/i });
    await search.fill('agent');
    await expect(page.getByText('Agent 42')).toBeVisible();
    await expect(page.getByText('Agént Diakritika')).toBeVisible();
  });

  test('US-51: Esc clears the search input', async ({ page }) => {
    await page.goto('/clients');
    const search = page.getByRole('searchbox', { name: /hledat/i });
    await search.fill('agent');
    await search.press('Escape');
    await expect(search).toHaveValue('');
  });
});

test.describe('clients page — drag reorder', () => {
  test('US-52: dragging a client to a new position persists across reload', async ({ page }) => {
    await page.goto('/clients');

    const handles = page.getByRole('button', { name: /přetáhnout pro změnu pořadí/i });
    const handleA = handles.first();
    const handleB = handles.nth(1);

    const handleABox = await handleA.boundingBox();
    const handleBBox = await handleB.boundingBox();
    if (!handleABox || !handleBBox) throw new Error('handles not visible');

    await handleA.hover();
    await page.mouse.down();
    await page.mouse.move(
      handleBBox.x + handleBBox.width / 2,
      handleBBox.y + handleBBox.height + 5,
      { steps: 10 },
    );
    await page.mouse.up();

    // After the drop, B comes before A
    const rows = page.locator('ul > li').filter({ hasText: /Agent|Agént/ });
    await expect(rows.first()).toContainText('Agént');

    await page.reload();
    const rowsAfter = page.locator('ul > li').filter({ hasText: /Agent|Agént/ });
    await expect(rowsAfter.first()).toContainText('Agént');
  });

  test('US-52: optimistic update reverts on server error', async ({ page }) => {
    await page.route('**/clients', async (route) => {
      // Intercept only the server-action POST (not the page nav). Match on the
      // request method + content-type; pass others through.
      const req = route.request();
      if (
        req.method() === 'POST' &&
        (req.headers()['content-type'] ?? '').includes('text/x-component')
      ) {
        await route.fulfill({ status: 500, body: 'fail' });
      } else {
        await route.continue();
      }
    });
    await page.goto('/clients');
    const handles = page.getByRole('button', { name: /přetáhnout/i });
    const a = handles.first();
    const b = handles.nth(1);
    const aBox = await a.boundingBox();
    const bBox = await b.boundingBox();
    if (!aBox || !bBox) throw new Error('handles not visible');
    await a.hover();
    await page.mouse.down();
    await page.mouse.move(bBox.x + bBox.width / 2, bBox.y + bBox.height + 5, { steps: 10 });
    await page.mouse.up();

    // Order snaps back; error banner renders
    await expect(page.getByText(/nepodařilo se|chyba/i)).toBeVisible();
    const rows = page.locator('ul > li').filter({ hasText: /Agent|Agént/ });
    await expect(rows.first()).toContainText('Agent 42');
  });

  test('US-52: keyboard reorder persists across reload', async ({ page }) => {
    await page.goto('/clients');
    const firstHandle = page.getByRole('button', { name: /přetáhnout pro změnu pořadí/i }).first();
    await firstHandle.focus();
    await page.keyboard.press('Space');
    await page.keyboard.press('ArrowDown');
    await page.keyboard.press('Space');

    const rows = page.locator('ul > li').filter({ hasText: /Agent|Agént/ });
    await expect(rows.first()).toContainText('Agént');

    await page.reload();
    const rowsAfter = page.locator('ul > li').filter({ hasText: /Agent|Agént/ });
    await expect(rowsAfter.first()).toContainText('Agént');
  });

  test('US-53: dragging a project within a client persists across reload', async ({ page }) => {
    await page.goto('/clients');
    await page.getByRole('button', { name: 'Rozbalit projekty' }).first().click();

    const projectHandles = page
      .locator('ul ul')
      .first()
      .getByRole('button', { name: /přetáhnout pro změnu pořadí/i });

    const first = projectHandles.first();
    const second = projectHandles.nth(1);
    const secondBox = await second.boundingBox();
    if (!secondBox) throw new Error('project handles missing');
    await first.hover();
    await page.mouse.down();
    await page.mouse.move(secondBox.x + secondBox.width / 2, secondBox.y + secondBox.height + 5, {
      steps: 10,
    });
    await page.mouse.up();

    const projects = page.locator('ul ul').first().locator('li');
    await expect(projects.first()).toContainText('Instalace agenta');

    await page.reload();
    await page.getByRole('button', { name: 'Rozbalit projekty' }).first().click();
    const projectsAfter = page.locator('ul ul').first().locator('li');
    await expect(projectsAfter.first()).toContainText('Instalace agenta');
  });

  test('US-53: drag handles are hidden and a hint shows while a search is active', async ({
    page,
  }) => {
    await page.goto('/clients');
    const search = page.getByRole('searchbox', { name: /hledat/i });
    await search.fill('agent');
    await expect(page.getByText(/vyhledávání je aktivní – zrušte ho pro řazení/i)).toBeVisible();

  });

test.describe('clients page — touch', () => {
  test.use({ ...devices['iPhone 13'] });
  test('US-52: touch drag reorders clients', async ({ page }) => {
    await page.goto('/clients');
    const handles = page.getByRole('button', { name: /přetáhnout/i });
    const a = handles.first();
    const b = handles.nth(1);
    const aBox = await a.boundingBox();
    const bBox = await b.boundingBox();
    if (!aBox || !bBox) throw new Error('handles not visible');

    // Pointer events with touch type — dnd-kit's PointerSensor handles these.
    const startX = aBox.x + aBox.width / 2;
    const startY = aBox.y + aBox.height / 2;
    const endX = bBox.x + bBox.width / 2;
    const endY = bBox.y + bBox.height + 8;

    await page.touchscreen.tap(startX, startY); // ensure focus
    await a.dispatchEvent('pointerdown', {
      pointerType: 'touch',
      clientX: startX,
      clientY: startY,
    });
    await a.dispatchEvent('pointermove', { pointerType: 'touch', clientX: endX, clientY: endY });
    await a.dispatchEvent('pointerup', { pointerType: 'touch', clientX: endX, clientY: endY });

    const rows = page.locator('ul > li').filter({ hasText: /Agent|Agént/ });
    await expect(rows.first()).toContainText('Agént');
  });
});
